QUERY COMUNI:
━━━━━━━━━━━━

1. Recupera tutti i record di employees per un tenant:
{
  "table": "employees",
  "select": [
    "id",
    "employee_code",
    "first_name",
    "last_name",
    "email"
  ],
  "where": {
    "tenant_id": "uuid_tenant"
  },
  "limit": 100
}

2. Recupera employees con dati di departments:
{
  "table": "employees",
  "select": [
    "*"
  ],
  "leftJoin": [
    {
      "table": "departments",
      "first": "employees.department_id",
      "second": "departments.id"
    }
  ],
  "limit": 50
}

3. Ricerca employees per first_name:
{
  "table": "employees",
  "select": [
    "*"
  ],
  "where": {
    "first_name": {
      "operator": "like",
      "value": "%search%"
    }
  },
  "limit": 20
}

4. Conta employees creati nell'ultimo mese:
{
  "table": "employees",
  "select": [
    "COUNT(*) as total"
  ],
  "where": {
    "created_at": {
      "operator": ">=",
      "value": "NOW() - INTERVAL '30 days'"
    }
  }
}
